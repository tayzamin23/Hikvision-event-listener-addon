Import requests, time, urllib3
from requests.auth import HTTPDigestAuth

urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

CAMERA_IP = "192.168.110.74"
USERNAME = "admin"
PASSWORD = "Diginet@2025"
WEBHOOK_URL = "http://192.168.110.26:8123/api/webhook/hikvision_event"

url = f"https://{CAMERA_IP}/ISAPI/Event/notification/alertStream"

session = requests.Session()

headers = {
    "User-Agent": "Mozilla/5.0",
    "Accept": "*/*",
    "Connection": "keep-alive",
    "Cache-Control": "no-cache",
    "Pragma": "no-cache"
}

while True:
    try:
        print("Connecting to Hikvision event stream...")

        r = session.get(
            url,
            auth=HTTPDigestAuth(USERNAME, PASSWORD),
            stream=True,
            timeout=20,
            verify=False,
            headers=headers
        )

        print("Connected. Waiting for events...")

        for raw_line in r.iter_lines():
            if not raw_line:
 		continue

            # FIX: convert bytes → str safely
            try:
                line = raw_line.decode(errors="ignore")
            except:
                line = str(raw_line)

            if "<eventType>" in line:
                print("EVENT:", line)

                if "face" in line.lower():
                    print("FACE DETECTED → Sending to Home Assistant")
                    requests.post(WEBHOOK_URL, json={"eventType": "faceDetection"})

    except Exception as e:
        print("Connection lost:", e)
        time.sleep(5)
